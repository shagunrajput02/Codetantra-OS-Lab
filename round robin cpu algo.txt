#include<stdio.h>
#include<conio.h>
void main()
{
	int i, NOP,sum=0,count=0,y,quant,wt=0,tat=0,at[10],bt[10],temp[10];
	float avg_wt, avg_tat;
	printf("Enter Total Number of Processes: ");
	scanf("%d",&NOP);
	y=NOP;
	for(i=0;i<NOP;i++){
		printf("Enter Details of Process[%d]: Arrival Time:\t",i+1);
		scanf("%d",&at[i]);
		printf("Burst Time:\t");
		scanf("%d",&bt[i]);
		temp[i]=bt[i];
	}
	printf("Enter Time Quantum:\t");
	scanf("%d",&quant);
	printf("Process ID\t\tBurst Time\t Turnaround Time\t Waiting Time\n");
	for(sum=0,i=0;y!=0;){
		if(temp[i]<=quant && temp[i]>0){
			sum+=temp[i];
			temp[i]=0;
			count=1;
		}
		else if(temp[i]>0){
			temp[i]=temp[i]-quant;
			sum+=quant;
		}
		if(temp[i]==0 && count==1){
			y--;
			printf("Process[%d]\t\t%d\t\t %d\t\t\t %d\n",i+1,bt[i],sum-at[i],sum-at[i]-bt[i]);
			wt=wt+sum-at[i]-bt[i];
			tat=tat+sum-at[i];
			count=0;
		}
		if(i==NOP-1){
			i=0;
		}
		else if(at[i+1]<=sum){
			i++;
		}
		else{
			i=0;
		}
	}
	avg_wt=(float)wt/NOP;
	avg_tat=(float)tat/NOP;
	printf("Average Waiting Time:\t%f\n",avg_wt);
	printf("Avg Turnaround Time:\t%f\n",avg_tat);
}
